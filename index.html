Factorial and Fibonacci
Iterative factorial and iterative Fibonacci
// MathExamples.java
public class MathExamples {
    // iterative factorial
    public static long factorial(int n) {
        long f = 1;
        for (int i = 2; i <= n; i++) f *= i;
        return f;
    }

    // iterative fibonacci (n-th, 0-based)
    public static long fibonacci(int n) {
        if (n <= 1) return n;
        long a = 0, b = 1;
        for (int i = 2; i <= n; i++) {
            long t = a + b;
            a = b;
            b = t;
        }
        return b;
    }

    public static void main(String[] args) {
        System.out.println("5! = " + factorial(5));
        System.out.println("Fibonacci(10) = " + fibonacci(10));
    }
}
Prime Check
Efficient check up to sqrt(n)
// PrimeCheck.java
public class PrimeCheck {
    public static boolean isPrime(int n) {
        if (n <= 1) return false;
        if (n <= 3) return true;
        if (n % 2 == 0 || n % 3 == 0) return false;
        for (int i = 5; i * i <= n; i += 6) {
            if (n % i == 0 || n % (i + 2) == 0) return false;
        }
        return true;
    }

    public static void main(String[] args) {
        int[] tests = {1,2,3,4,17,18,19,97};
        for (int t : tests) {
            System.out.println(t + " is prime? " + isPrime(t));
        }
    }
}

Simple Interest Calculator
Formula: SI = (P * R * T) / 100
// SimpleInterest.java
import java.util.Scanner;

public class SimpleInterest {
    public static double calculateInterest(double principal, double rate, double time) {
        return (principal * rate * time) / 100;
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter Principal amount (P): ");
        double principal = sc.nextDouble();
        System.out.print("Enter Rate of interest (R): ");
        double rate = sc.nextDouble();
        System.out.print("Enter Time period in years (T): ");
        double time = sc.nextDouble();

        double interest = calculateInterest(principal, rate, time);
        double totalAmount = principal + interest;

        System.out.printf("Simple Interest = %.2f%n", interest);
        System.out.printf("Total Amount = %.2f%n", totalAmount);
        sc.close();
    }
}

Quadratic Equation Solver (ax^2 + bx + c = 0)
Finds real and complex roots using discriminant
// QuadraticEquation.java
import java.util.Scanner;

public class QuadraticEquation {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.println("Solve quadratic equation: ax*ax + bx + c = 0");
        System.out.print("Enter a: ");
        double a = sc.nextDouble();
        System.out.print("Enter b: ");
        double b = sc.nextDouble();
        System.out.print("Enter c: ");
        double c = sc.nextDouble();

        if (a == 0) {
            System.out.println("This is a linear equation (a cannot be 0).");
            if (b != 0) {
                double root = -c / b;
                System.out.printf("Root: x = %.2f%n", root);
            } else {
                System.out.println(c == 0 ? "Infinite solutions" : "No solution");
            }
        } else {
            double discriminant = b * b - 4 * a * c;
            if (discriminant > 0) {
                double root1 = (-b + Math.sqrt(discriminant)) / (2 * a);
                double root2 = (-b - Math.sqrt(discriminant)) / (2 * a);
                System.out.printf("Two distinct real roots: %.2f and %.2f%n", root1, root2);
            } else if (discriminant == 0) {
                double root = -b / (2 * a);
                System.out.printf("One real double root: %.2f%n", root);
            } else {
                double real = -b / (2 * a);
                double imaginary = Math.sqrt(-discriminant) / (2 * a);
                System.out.printf("Complex roots: %.2f + %.2fi and %.2f - %.2fi%n",
                                  real, imaginary, real, imaginary);
            }
        }
        sc.close();
    }
}
Compile: javac QuadraticEquation.java    Run: java QuadraticEquation
Second Smallest Number (No if-else, no ternary)
Uses only Math.max() and Math.min() and accepts three unequal numbers
// SecondSmallest.java
import java.util.Scanner;

public class SecondSmallest {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter three unequal numbers: ");
        double a = sc.nextDouble();
        double b = sc.nextDouble();
        double c = sc.nextDouble();

        double max = Math.max(a, Math.max(b, c));
        double min = Math.min(a, Math.min(b, c));
        double secondSmallest = a + b + c - max - min;
        
        System.out.println("Second smallest number: " + secondSmallest);
        sc.close();
    }
}

Student Grade Calculator
Input name, class, marks in English, Mathematics, Computer ? average & grade
// StudentGrade.java
import java.util.Scanner;

public class StudentGrade {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter student name: ");
        String name = sc.nextLine();
        System.out.print("Enter class: ");
        String className = sc.nextLine();
        System.out.print("Enter marks in English (out of 100): ");
        double eng = sc.nextDouble();
        System.out.print("Enter marks in Mathematics (out of 100): ");
        double math = sc.nextDouble();
        System.out.print("Enter marks in Computer (out of 100): ");
        double comp = sc.nextDouble();
        
        double average = (eng + math + comp) / 3;
        String grade;
        if (average >= 80) {
            grade = "A*";
        } else if (average >= 60) {
            grade = "A";
        } else if (average >= 50) {
            grade = "B";
        } else if (average >= 40) {
            grade = "C";
        } else {
            grade = "Fail";
        }
        
        System.out.println("\n--- Result ---");
        System.out.println("Name: " + name);
        System.out.println("Class: " + className);
        System.out.printf("Average marks: %.2f%%%n", average);
        System.out.println("Grade: " + grade);
        sc.close();
    }
}
Compile: javac StudentGrade.java    Run: java StudentGrade

        return 0.5 * base * height;
    }
    public static void main(String[] args) {
        double base = 6.0;
        double height = 4.0;
        System.out.println("Triangle base = " + base + ", height = " + height);
        System.out.println("Area = " + area(base, height));
    }

// QuadrilateralArea.java
public class QuadrilateralArea {
    public static double area(double[][] pts) {
        double sum = 0.0;
        for (int i = 0; i < 4; i++) {
            int j = (i + 1) % 4;
            sum += pts[i][0] * pts[j][1] - pts[j][0] * pts[i][1];
        }
        return Math.abs(sum) / 2.0;
    }
    public static void main(String[] args) {
        double[][] quad = {{0.0,0.0},{4.0,0.0},{4.0,3.0},{0.0,3.0}};
        System.out.println("Quadrilateral Area = " + area(quad));
    }
}
File Input and Output
Writes and reads a small text file
// FileIOExample.java
import java.io.IOException;
import java.nio.file.Files;
import java.nio.file.Path;
import java.nio.charset.StandardCharsets;

public class FileIOExample {
    public static void main(String[] args) {
        Path p = Path.of("sample.txt");
        String text = "Hello from Java World\nLine two";
        try {
            Files.writeString(p, text, StandardCharsets.UTF_8);
            String read = Files.readString(p, StandardCharsets.UTF_8);
            System.out.println("File contents:\n" + read);
        } catch (IOException e) {
            System.err.println("IO error: " + e.getMessage());
        }
    }
}
Exception Handling Example
Demonstrates try-catch-finally
// ExceptionDemo.java
public class ExceptionDemo {
    public static int divide(int a, int b) {
        try {
            return a / b;
        } catch (ArithmeticException ex) {
            System.err.println("Cannot divide by zero");
            return 0;
        } finally {
            System.out.println("Division attempt finished");
        }
    }
    public static void main(String[] args) {
        System.out.println(divide(10, 2));
        System.out.println(divide(10, 0));
    }
}
