import java.util.Scanner;

public class Diamond {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter half-size (odd n): ");
        int n = sc.nextInt();
        // upper half
        for (int i = 1; i <= n; i += 2) {
            for (int s = (n - i) / 2; s > 0; s--) System.out.print(" ");
            for (int j = 1; j <= i; j++) System.out.print("*");
            System.out.println();
        }
        // lower half
        for (int i = n - 2; i >= 1; i -= 2) {
            for (int s = (n - i) / 2; s > 0; s--) System.out.print(" ");
            for (int j = 1; j <= i; j++) System.out.print("*");
            System.out.println();
        }
        sc.close();
    }
}

Palindrome Check (String)
// PalindromeCheck.java
import java.util.Scanner;

public class PalindromeCheck {
    public static boolean isPalindrome(String s) {
        s = s.toLowerCase().replaceAll("[^a-z0-9]", "");
        int l = 0, r = s.length() - 1;
        while (l < r) {
            if (s.charAt(l) != s.charAt(r)) return false;
            l++; r--;
        }
        return true;
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter a word or sentence: ");
        String input = sc.nextLine();
        System.out.println(isPalindrome(input) ? "\"" + input + "\" IS a palindrome"
                                               : "\"" + input + "\" is NOT a palindrome");
        sc.close();
    }
}

Reverse a String
// ReverseString.java
import java.util.Scanner;

public class ReverseString {
    // Using StringBuilder
    public static String reverse(String s) {
        return new StringBuilder(s).reverse().toString();
    }

    // Manual method
    public static String reverseManual(String s) {
        char[] ch = s.toCharArray();
        int l = 0, r = ch.length - 1;
        while (l < r) {
            char tmp = ch[l];
            ch[l++] = ch[r];
            ch[r--] = tmp;
        }
        return new String(ch);
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter a string: ");
        String input = sc.nextLine();
        System.out.println("Reversed (StringBuilder): " + reverse(input));
        System.out.println("Reversed (manual)       : " + reverseManual(input));
        sc.close();
    }
}

Anagram Check
// AnagramCheck.java
import java.util.Arrays;
import java.util.Scanner;

public class AnagramCheck {
    public static boolean isAnagram(String a, String b) {
        a = a.toLowerCase().replaceAll("\\s", "");
        b = b.toLowerCase().replaceAll("\\s", "");
        if (a.length() != b.length()) return false;
        char[] ca = a.toCharArray();
        char[] cb = b.toCharArray();
        Arrays.sort(ca);
        Arrays.sort(cb);
        return Arrays.equals(ca, cb);
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter first word : ");  String a = sc.nextLine();
        System.out.print("Enter second word: ");  String b = sc.nextLine();
        System.out.println(isAnagram(a, b) ? "\"" + a + "\" and \"" + b + "\" ARE anagrams"
                                           : "NOT anagrams");
        sc.close();
    }
}

// ArmstrongNumber.java
import java.util.Scanner;

public class ArmstrongNumber {
    public static boolean isArmstrong(int n) {
        int original = n, digits = String.valueOf(n).length(), sum = 0;
        while (n > 0) {
            int d = n % 10;
            sum += (int) Math.pow(d, digits);
            n /= 10;
        }
        return sum == original;
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter a number: ");
        int n = sc.nextInt();
        System.out.println(n + (isArmstrong(n) ? " IS" : " is NOT") + " an Armstrong number");

        // Print all 3-digit Armstrong numbers
        System.out.print("3-digit Armstrong numbers: ");
        for (int i = 100; i <= 999; i++) {
            if (isArmstrong(i)) System.out.print(i + " ");
        }
        System.out.println();
        sc.close();
    }
}

import java.util.Scanner;

public class LeapYear {
    public static boolean isLeap(int year) {
        return (year % 400 == 0) || (year % 4 == 0 && year % 100 != 0);
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter a year: ");
        int year = sc.nextInt();
        System.out.println(year + (isLeap(year) ? " IS a leap year" : " is NOT a leap year"));

        // Next 5 leap years
        System.out.print("Next 5 leap years from " + year + ": ");
        int count = 0;
        for (int y = year + 1; count < 5; y++) {
            if (isLeap(y)) { System.out.print(y + " "); count++; }
        }
        System.out.println();
        sc.close();
    }

GCD and LCM
// GcdLcm.java
import java.util.Scanner;

public class GcdLcm {
    public static int gcd(int a, int b) {
        while (b != 0) {
            int t = b;
            b = a % b;
            a = t;
        }
        return a;
    }

    public static long lcm(int a, int b) {
        return (long) a / gcd(a, b) * b;   // avoid overflow
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter two positive integers: ");
        int a = sc.nextInt(), b = sc.nextInt();
        System.out.println("GCD(" + a + ", " + b + ") = " + gcd(a, b));
        System.out.println("LCM(" + a + ", " + b + ") = " + lcm(a, b));
        sc.close();
    }
}
Power Without Math.pow
// PowerCalc.java
import java.util.Scanner;

public class PowerCalc {
    // Fast exponentiation (exponentiation by squaring)
    public static long power(long base, int exp) {
        long result = 1;
        while (exp > 0) {
            if (exp % 2 == 1) result *= base;
            base *= base;
            exp /= 2;
        }
        return result;
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter base: ");
        long base = sc.nextLong();
        System.out.print("Enter exponent (non-negative): ");
        int exp = sc.nextInt();
        System.out.println(base + "^" + exp + " = " + power(base, exp));
        sc.close();
    }
}

import java.util.Scanner;

public class TempConversion {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.println("Convert from: 1) Celsius  2) Fahrenheit  3) Kelvin");
        System.out.print("Choice: ");
        int choice = sc.nextInt();
        System.out.print("Enter temperature: ");
        double temp = sc.nextDouble();

        switch (choice) {
            case 1:
                System.out.printf("%.2f°C = %.2f°F%n", temp, temp * 9/5 + 32);
                System.out.printf("%.2f°C = %.2fK%n",  temp, temp + 273.15);
                break;
            case 2:
                System.out.printf("%.2f°F = %.2f°C%n", temp, (temp - 32) * 5/9);
                System.out.printf("%.2f°F = %.2fK%n",  temp, (temp - 32) * 5/9 + 273.15);
                break;
            case 3:
                System.out.printf("%.2fK = %.2f°C%n", temp, temp - 273.15);
                System.out.printf("%.2fK = %.2f°F%n", temp, (temp - 273.15) * 9/5 + 32);
                break;
            default:
                System.out.println("Invalid choice");
        }
        sc.close();
    }
}

        System.out.println("ax² + bx + c = 0");

        System.out.print("Enter b: "); double b = sc.nextDouble();
        System.out.print("Enter c: "); double c = sc.nextDouble();

        double disc = b * b - 4 * a * c;
        if (disc > 0) {
            double r1 = (-b + Math.sqrt(disc)) / (2 * a);
            double r2 = (-b - Math.sqrt(disc)) / (2 * a);
            System.out.printf("Two real roots: %.4f and %.4f%n", r1, r2);
        } else if (disc == 0) {
            System.out.printf("One real root: %.4f%n", -b / (2 * a));
        } else {
            double re = -b / (2 * a);
            double im = Math.sqrt(-disc) / (2 * a);
            System.out.printf("Complex roots: %.4f + %.4fi  and  %.4f - %.4fi%n", re, im, re, im);
        }
        sc.close();
    }

import java.util.Arrays;

public class ArrayReverse {
    public static void reverseInPlace(int[] arr) {
        int l = 0, r = arr.length - 1;
        while (l < r) {
            int tmp = arr[l]; arr[l++] = arr[r]; arr[r--] = tmp;
        }
    }

    public static void main(String[] args) {
        int[] original = {10, 20, 30, 40, 50};
        System.out.println("Original  : " + Arrays.toString(original));

        // copy then reverse
        int[] copy = Arrays.copyOf(original, original.length);
        reverseInPlace(copy);
        System.out.println("Reversed  : " + Arrays.toString(copy));
        System.out.println("Original still: " + Arrays.toString(original));
    }

// BinarySearch.java
import java.util.Arrays;
import java.util.Scanner;

public class BinarySearch {
    public static int binarySearch(int[] arr, int key) {
        int low = 0, high = arr.length - 1;
        while (low <= high) {
            int mid = low + (high - low) / 2;
            if (arr[mid] == key) return mid;
            else if (arr[mid] < key) low = mid + 1;
            else high = mid - 1;
        }
        return -1;
    }

    public static void main(String[] args) {
        int[] arr = {2, 5, 8, 12, 16, 23, 38, 56, 72, 91};
        System.out.println("Sorted array: " + Arrays.toString(arr));
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter key: ");
        int key = sc.nextInt();
        int idx = binarySearch(arr, key);
        System.out.println(idx >= 0 ? key + " found at index " + idx : key + " not found");
        sc.close();
    }

// BubbleSort.java
import java.util.Arrays;

public class BubbleSort {
    public static void bubbleSort(int[] arr) {
        int n = arr.length;
        for (int i = 0; i < n - 1; i++) {
            boolean swapped = false;
            for (int j = 0; j < n - 1 - i; j++) {
                if (arr[j] > arr[j + 1]) {
                    int tmp = arr[j]; arr[j] = arr[j+1]; arr[j+1] = tmp;
                    swapped = true;
                }
            }
            if (!swapped) break;  // already sorted
        }
    }

    public static void main(String[] args) {
        int[] arr = {64, 34, 25, 12, 22, 11, 90};
        System.out.println("Before: " + Arrays.toString(arr));
        bubbleSort(arr);
        System.out.println("After : " + Arrays.toString(arr));
    }
}

    public static void selectionSort(int[] arr) {
        int n = arr.length;
        for (int i = 0; i < n - 1; i++) {
            int minIdx = i;
            for (int j = i + 1; j < n; j++)

        }
    }

import java.util.Arrays;

public class InsertionSort {
    public static void insertionSort(int[] arr) {
        for (int i = 1; i < arr.length; i++) {
            int key = arr[i];
            int j = i - 1;
            while (j >= 0 && arr[j] > key) {
                arr[j + 1] = arr[j];
                j--;
            }
            arr[j + 1] = key;
        }
    }

    public static void main(String[] args) {
        int[] arr = {12, 11, 13, 5, 6};
        System.out.println("Before: " + Arrays.toString(arr));
        insertionSort(arr);

        long f = 1;

        if (n <= 3) return true;
        if (n % 2 == 0 || n % 3 == 0) return false;

import java.util.Arrays;

public class SieveOfEratosthenes {
    public static void main(String[] args) {
        int limit = 100;
        boolean[] isComposite = new boolean[limit + 1];
        Arrays.fill(isComposite, false);

        for (int i = 2; i * i <= limit; i++) {
            if (!isComposite[i]) {
                for (int j = i * i; j <= limit; j += i)
                    isComposite[j] = true;
            }

        System.out.println("Primes up to " + limit + ":");
        int count = 0;
        for (int i = 2; i <= limit; i++) {
            if (!isComposite[i]) { System.out.printf("%4d", i); count++; }
        }
        System.out.println("\nTotal: " + count + " primes");
    }
}

    }

// TriangleArea.java
public class TriangleArea {
    // base & height
    static double areaBaseHeight(double b, double h) { return 0.5 * b * h; }
    // Heron's formula (three sides)
    static double areaHeron(double a, double b, double c) {
        double s = (a + b + c) / 2;
        return Math.sqrt(s * (s-a) * (s-b) * (s-c));
    }
    public static void main(String[] args) {
        System.out.println("Triangle (b=6, h=4)  area = " + areaBaseHeight(6, 4));
        System.out.println("Triangle (3,4,5 sides) area = " + areaHeron(3, 4, 5));
    }
}
// QuadrilateralArea.java
public class QuadrilateralArea {
    public static double area(double[][] pts) {
        double sum = 0;
        for (int i = 0; i < 4; i++) {
            int j = (i + 1) % 4;
            sum += pts[i][0]*pts[j][1] - pts[j][0]*pts[i][1];
        }
        return Math.abs(sum) / 2.0;
    }
    public static void main(String[] args) {
        double[][] quad = {{0,0},{4,0},{4,3},{0,3}};
        System.out.println("Quadrilateral area = " + area(quad));
    }
}

Exception Handling

try / catch / finally
// ExceptionDemo.java
public class ExceptionDemo {
    public static int divide(int a, int b) {
        try {
            return a / b;
        } catch (ArithmeticException ex) {
            System.err.println("Error: " + ex.getMessage());
            return 0;
        } finally {
            System.out.println("Division attempt finished");
        }
    }
    public static void main(String[] args) {
        System.out.println(divide(10, 2));
        System.out.println(divide(10, 0));
    }
}
Multiple Exceptions & Custom Exception
// CustomExceptionDemo.java

// Custom checked exception
class AgeException extends Exception {
    AgeException(String msg) { super(msg); }
}

public class CustomExceptionDemo {
    static void validateAge(int age) throws AgeException {
        if (age < 0 || age > 150)
            throw new AgeException("Invalid age: " + age);
    }

    static int parseAndDivide(String s, int divisor) {
        try {
            int value = Integer.parseInt(s);
            return value / divisor;
        } catch (NumberFormatException e) {
            System.out.println("Not a number: " + s);
        } catch (ArithmeticException e) {
            System.out.println("Division by zero!");
        }
        return 0;
    }

    public static void main(String[] args) {
        // custom exception
        int[] ages = {25, -1, 200};
        for (int age : ages) {
            try {
                validateAge(age);
                System.out.println(age + " is valid.");
            } catch (AgeException e) {
                System.out.println("Caught: " + e.getMessage());
            }
        }
        // multiple catch types
        System.out.println(parseAndDivide("100", 4));
        System.out.println(parseAndDivide("abc", 4));
        System.out.println(parseAndDivide("100", 0));
    }
}
