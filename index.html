Palindrome Checker
Check whether a number is palindrome (e.g., 121 → Palindrome)
// PalindromeNumber.java
import java.util.Scanner;
public class PalindromeNumber {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter an integer: ");
        int num = sc.nextInt();
        int original = num;
        int reversed = 0;
        while (num != 0) {
            int digit = num % 10;
            reversed = reversed * 10 + digit;
            num /= 10;
        }
        if (original == reversed)
            System.out.println(original + " is a palindrome.");
        else
            System.out.println(original + " is not a palindrome.");
        sc.close();
    }
}
Example: 121 → Palindrome
Armstrong Number Checker
Check whether a number is an Armstrong number (e.g., 153 = 1³ + 5³ + 3³)
// ArmstrongNumber.java
import java.util.Scanner;
public class ArmstrongNumber {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter an integer: ");
        int num = sc.nextInt();
        int original = num;
        int sum = 0;
        int digits = String.valueOf(num).length();
        int temp = num;
        while (temp != 0) {
            int digit = temp % 10;
            sum += Math.pow(digit, digits);
            temp /= 10;
        }
        if (sum == original)
            System.out.println(original + " is an Armstrong number.");
        else
            System.out.println(original + " is not an Armstrong number.");
        sc.close();
    }
}
Example: 153 → 1³+5³+3³ = 153  |  Compile: javac ArmstrongNumber.java

Factorial and Fibonacci
Iterative
// MathExamples.java
public class MathExamples {
    public static long factorial(int n) {
        long f = 1;
        for (int i = 2; i <= n; i++) f *= i;
        return f;
    }
    public static long fibonacci(int n) {
        if (n <= 1) return n;
        long a = 0, b = 1;
        for (int i = 2; i <= n; i++) {
            long t = a + b;
            a = b;
            b = t;
        }
        return b;
    }
    public static void main(String[] args) {
        System.out.println("5! = " + factorial(5));
        System.out.println("Fib(10) = " + fibonacci(10));
    }
}
Prime Check
Efficient sqrt(n)
// PrimeCheck.java
public class PrimeCheck {
    public static boolean isPrime(int n) {
        if (n <= 1) return false;
        if (n <= 3) return true;
        if (n % 2 == 0 || n % 3 == 0) return false;
        for (int i = 5; i * i <= n; i += 6) {
            if (n % i == 0 || n % (i + 2) == 0) return false;
        }
        return true;
    }
    public static void main(String[] args) {
        System.out.println("17 is prime? " + isPrime(17));
    }
}

Quadratic Equation Solver
ax² + bx + c = 0
// QuadraticEquation.java
import java.util.Scanner;
public class QuadraticEquation {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("a b c: ");
        double a = sc.nextDouble();
        double b = sc.nextDouble();
        double c = sc.nextDouble();
        double d = b * b - 4 * a * c;
        if (d > 0) {
            double r1 = (-b + Math.sqrt(d)) / (2 * a);
            double r2 = (-b - Math.sqrt(d)) / (2 * a);
            System.out.println("Roots: " + r1 + " and " + r2);
        } else if (d == 0) {
            double r = -b / (2 * a);
            System.out.println("Root: " + r);
        } else {
            double real = -b / (2 * a);
            double imag = Math.sqrt(-d) / (2 * a);
            System.out.println("Complex: " + real + " ± i" + imag);
        }
        sc.close();
    }
}

Second Smallest (no if/ternary)
Using Math.min/max
// SecondSmallest.java
import java.util.Scanner;
public class SecondSmallest {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter three unequal numbers: ");
        double a = sc.nextDouble();
        double b = sc.nextDouble();
        double c = sc.nextDouble();
        double max = Math.max(a, Math.max(b, c));
        double min = Math.min(a, Math.min(b, c));
        double second = a + b + c - max - min;
        System.out.println("Second smallest number: " + second);
        sc.close();
    }
}
Sample: 34 82 61 → 61.0
Student Grade Calculator
Average & grade table
// StudentGrade.java
import java.util.Scanner;
public class StudentGrade {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Name: ");
        String name = sc.nextLine();
        System.out.print("Class: ");
        String cls = sc.nextLine();
        System.out.print("Marks in Eng, Math, Comp: ");
        double e = sc.nextDouble();
        double m = sc.nextDouble();
        double c = sc.nextDouble();
        double avg = (e + m + c) / 3;
        String grade;
        if (avg >= 80) grade = "A*";
        else if (avg >= 60) grade = "A";
        else if (avg >= 50) grade = "B";
        else if (avg >= 40) grade = "C";
        else grade = "Fail";
        System.out.printf("Name: %s, Class: %s, Avg: %.2f, Grade: %s", name, cls, avg, grade);
        sc.close();
    }
}

// QuadrilateralArea.java
public class QuadrilateralArea {
    public static double area(double[][] pts) {
        double s = 0;
        for (int i = 0; i < 4; i++) {
            int j = (i + 1) % 4;
            s += pts[i][0] * pts[j][1] - pts[j][0] * pts[i][1];
        }
        return Math.abs(s) / 2.0;
    }
    public static void main(String[] args) {
        double[][] q = {{0, 0}, {4, 0}, {4, 3}, {0, 3}};
        System.out.println("Area: " + area(q));
    }
}

Exception Handling
try-catch-finally
// ExceptionDemo.java
public class ExceptionDemo {
    public static int divide(int a, int b) {
        try {
            return a / b;
        } catch (ArithmeticException e) {
            System.err.println("Divide by zero");
            return 0;
        } finally {
            System.out.println("Attempt done");
        }
    }
    public static void main(String[] args) {
        System.out.println(divide(10, 2));
        System.out.println(divide(10, 0));
    }
}
